19.2 Case Study 1: Single-Dose Oral Antibiotic

19.2.1 Background

A new oral antibiotic was administered to 12 healthy volunteers (500 mg tablet). Blood samples were collected at 0.5, 1, 1.5, 2, 3, 4, 6, 8, 12, and 24 hours post-dose.

Code 
library(ggplot2)
library(dplyr)
library(rxode2)

# Simulate the dataset
set.seed(123)
n_subj <- 12
times <- c(0.5, 1, 1.5, 2, 3, 4, 6, 8, 12, 24)

# Population parameters
pop_ka <- 0.8   # 1/h
pop_CL <- 6.0   # L/h
pop_V  <- 55    # L

cs1_data <- expand.grid(ID = 1:n_subj, time = times) %>%
  mutate(
    eta_ka = rep(rnorm(n_subj, 0, 0.3), each = length(times)),
    eta_CL = rep(rnorm(n_subj, 0, 0.25), each = length(times)),
    eta_V  = rep(rnorm(n_subj, 0, 0.15), each = length(times)),
    ka = pop_ka * exp(eta_ka),
    CL = pop_CL * exp(eta_CL),
    V  = pop_V * exp(eta_V),
    ke = CL / V,
    DOSE = 500,
    conc = (DOSE * ka) / (V * (ka - ke)) * (exp(-ke * time) - exp(-ka * time)),
    conc_obs = conc * exp(rnorm(n(), 0, 0.15))
  ) %>%
  filter(conc_obs > 0)

# EDA: individual profiles
ggplot(cs1_data, aes(time, conc_obs, group = ID, color = factor(ID))) +
  geom_point(size = 1.5) +
  geom_line(alpha = 0.5) +
  labs(title = "Case Study 1: Individual PK Profiles",
       subtitle = "500 mg oral antibiotic, 12 subjects",
       x = "Time (hours)", y = "Concentration (mg/L)") +
  theme_minimal() +
  theme(legend.position = "none")

19.2.3 Compartmental Model Fit

Code 
# Fit one-compartment oral model to mean data
cs1_mean <- cs1_data %>%
  group_by(time) %>%
  summarise(conc = mean(conc_obs), .groups = "drop")

D <- 500  # administered dose (mg), known and fixed
fit_cs1 <- nls(conc ~ (D * ka) / (V * (ka - ke)) * (exp(-ke * time) - exp(-ka * time)),
               data = cs1_mean,

               lower = c(0.01, 0.01, 1),

19.3 Case Study 2: IV Infusion with Multiple Dosing

19.3.1 Background

A drug is administered as a 2-hour IV infusion every 8 hours. Target steady-state concentration: 10–20 mg/L.

Code 
# Simulate multiple IV infusions
mod_inf <- rxode2({
  d/dt(central) <- rate - ke * central
  conc <- central / V
})

# Regimen: 500 mg over 2h, q8h
dose_mg <- 500
inf_dur <- 2    # hours
tau <- 8        # hours
n_doses <- 6

ke <- 0.25; V <- 25

# Build event table
dose_times <- seq(0, (n_doses - 1) * tau, by = tau)
events <- do.call(rbind, lapply(dose_times, function(t) {
  data.frame(time = t, amt = dose_mg, rate = dose_mg / inf_dur, cmt = 1)
}))

times_cs2 <- seq(0, n_doses * tau, by = 0.05)
sim_cs2 <- rxSolve(mod_inf, c(ke = ke, V = V), et(events) %>% et(times_cs2))

cs2_df <- as.data.frame(sim_cs2)

# Therapeutic range visualization
ggplot(cs2_df, aes(time, conc)) +
  geom_line(color = "#0B6E6B", linewidth = 0.8) +
  geom_hline(yintercept = c(10, 20), linetype = "dashed",
             color = c("#D4A843", "#C44536")) +
  annotate("text", x = 40, y = c(10.5, 19.5),
           label = c("Lower target (10 mg/L)", "Upper target (20 mg/L)"),
           color = c("#D4A843", "#C44536"), size = 3, hjust = 0) +
  labs(title = "Case Study 2: Multiple IV Infusion (q8h)",
       subtitle = paste0("Dose: ", dose_mg, " mg over ", inf_dur, " h"),
       x = "Time (hours)", y = "Concentration (mg/L)") +
  theme_minimal()

Interpretation: Steady state is reached after ~5 half-lives (~14 h). All concentrations remain within the therapeutic range, suggesting this is an appropriate regimen.

19.4 Case Study 3: Antimicrobial PK/PD

19.4.1 Background

An antibiotic requires C_{min}/MIC > 1 and AUC/MIC > 125 for efficacy. MIC = 2 mg/L.

Code 
# PTA analysis for antimicrobial
set.seed(789)
n_sim <- 2000

# Population variability
pop_CL <- 5.0; omega_CL <- 0.25
pop_V  <- 45;  omega_V  <- 0.15
MIC <- 2  # mg/L

# Regimens to compare
regimens <- data.frame(
  Regimen = c("250 mg q12h", "500 mg q12h", "500 mg q8h", "1000 mg q12h"),
  Dose = c(250, 500, 500, 1000),
  Tau  = c(12, 12, 8, 12)
)

# Simulate PTA for each regimen
pta_results <- lapply(1:nrow(regimens), function(r) {
  CL_sim <- pop_CL * exp(rnorm(n_sim, 0, omega_CL))
  V_sim  <- pop_V  * exp(rnorm(n_sim, 0, omega_V))
  ke_sim <- CL_sim / V_sim

  dose <- regimens$Dose[r]
  tau  <- regimens$Tau[r]

  Cmin_ss <- (dose / V_sim) * exp(-ke_sim * tau) / (1 - exp(-ke_sim * tau))
  AUC_ss  <- dose / CL_sim

  data.frame(
    Regimen = regimens$Regimen[r],
    PTA_Cmin = mean(Cmin_ss > MIC),
    PTA_AUC_MIC = mean(AUC_ss / MIC > 125),
    Cmin_median = median(Cmin_ss),
    AUC_MIC_median = median(AUC_ss / MIC)
  )
}) %>% bind_rows()

19.5 Case Study 4: Delayed PD Response

19.5.1 Background

A drug shows delayed onset of effect — maximum response occurs after the plasma concentration peak. We model this with an effect compartment.

Code 
# Simulate PK with effect compartment
mod_ec <- rxode2({
  d/dt(depot)   <- -ka * depot
  d/dt(central) <-  ka * depot - ke * central
  d/dt(effect)  <- ke0 * (central/V - effect)
  Cp <- central / V
  Ce <- effect
  Response <- E0 + (Emax * Ce) / (EC50 + Ce)
})

params_cs4 <- c(ka = 1.2, ke = 0.3, V = 12,
                ke0 = 0.4, E0 = 10, Emax = 80, EC50 = 2)

event_cs4 <- et(time = 0, amt = 200)
times_cs4 <- seq(0, 24, by = 0.1)

sim_cs4 <- rxSolve(mod_ec, params_cs4, event_cs4 %>% et(times_cs4))
cs4_df <- as.data.frame(sim_cs4)

# Plot PK and PD
ggplot(cs4_df, aes(time)) +
  geom_line(aes(y = Cp, color = "Plasma Conc"), linewidth = 1) +
  geom_line(aes(y = Ce, color = "Effect Site Conc"), linewidth = 1, linetype = "dashed") +
  geom_line(aes(y = Response / 5, color = "Response (÷5)"), linewidth = 1.2) +
  scale_y_continuous(
    name = "Concentration (mg/L)",
    sec.axis = sec_axis(~ . * 5, name = "Response")
  ) +
  labs(title = "Case Study 4: Delayed PD Response",
       subtitle = paste0("Effect compartment model, ke0 = ", params_cs4["ke0"]),
       x = "Time (hours)", color = "") +
  scale_color_manual(values = c("#0B6E6B", "#D4A843", "#C44536")) +
  theme_minimal()

Interpretation: The effect compartment model captures the time delay between plasma concentration and response. The peak effect occurs ~1–2 hours after the plasma peak.

19.6 Case Study 5: Population PK with Covariates

19.6.1 Background

A population PK study in 80 patients with varying body weight (40–120 kg) and renal function (eGFR: 20–120 mL/min).

Code 
# Simulate population PK with covariates
set.seed(456)
n_pop <- 80
sparse_times <- c(1, 3, 8)  # 3 samples per subject

pop_data <- expand.grid(ID = 1:n_pop, time = sparse_times) %>%
  mutate(
    WT   = rep(rnorm(n_pop, 70, 15), each = length(sparse_times)),
    eGFR = rep(rnorm(n_pop, 80, 25), each = length(sparse_times)),

    # Covariate effects
    eta_CL = rep(rnorm(n_pop, 0, 0.2), each = length(sparse_times)),
    CL = 5.0 * (WT/70)^0.75 * (eGFR/90)^0.5 * exp(eta_CL),

    eta_V = rep(rnorm(n_pop, 0, 0.15), each = length(sparse_times)),
    V  = 50 * (WT/70)^1.0 * exp(eta_V),

    ke = CL / V,
    conc = (500/V) * exp(-ke * time),
    conc_obs = conc * exp(rnorm(n(), 0, 0.2))
  ) %>%
  arrange(ID, time)

# Visualize covariate effects
p1 <- ggplot(pop_data %>% distinct(ID, .keep_all = TRUE), aes(WT, CL)) +
  geom_point(alpha = 0.6, color = "#0B6E6B") +
  geom_smooth(method = "lm", se = TRUE, color = "#C44536") +
  labs(title = "Weight Effect on Clearance",
       x = "Body Weight (kg)", y = "CL (L/h)") +
  theme_minimal()

p2 <- ggplot(pop_data %>% distinct(ID, .keep_all = TRUE), aes(eGFR, CL)) +
  geom_point(alpha = 0.6, color = "#0B6E6B") +
  geom_smooth(method = "lm", se = TRUE, color = "#C44536") +
  labs(title = "Renal Function Effect on Clearance",
       x = "eGFR (mL/min)", y = "CL (L/h)") +
  theme_minimal()

library(patchwork)
p1 + p2

Interpretation: Both weight and renal function are significant covariates on CL. Failure to account for them would increase unexplained variability and could lead to suboptimal dosing.

19.7 Case Study 6: Dose Optimization by PTA

19.7.1 Background

Using the population model from Case Study 5, find the dose that achieves >90% PTA for target C_{min} > 2 mg/L.

Code 
# Dose optimization via simulation
dose_range <- seq(200, 2000, by = 50)
target_Cmin <- 2  # mg/L
tau <- 12  # hours

set.seed(999)
n_sim_opt <- 2000

# Simulate population
WT_sim   <- rnorm(n_sim_opt, 70, 15)
eGFR_sim <- rnorm(n_sim_opt, 80, 25)
eta_CL_sim <- rnorm(n_sim_opt, 0, 0.2)
eta_V_sim  <- rnorm(n_sim_opt, 0, 0.15)

CL_sim <- 5.0 * (WT_sim/70)^0.75 * (eGFR_sim/90)^0.5 * exp(eta_CL_sim)
V_sim  <- 50 * (WT_sim/70)^1.0 * exp(eta_V_sim)
ke_sim <- CL_sim / V_sim

# PTA vs dose
pta_doses <- sapply(dose_range, function(d) {
  Cmin_ss <- (d / V_sim) * exp(-ke_sim * tau) / (1 - exp(-ke_sim * tau))
  mean(Cmin_ss > target_Cmin)
})

df_pta_cs6 <- data.frame(Dose = dose_range, PTA = pta_doses)
optimal_dose <- dose_range[which.min(abs(pta_doses - 0.9))]

ggplot(df_pta_cs6, aes(Dose, PTA)) +
  geom_line(color = "#0B6E6B", linewidth = 1.2) +
  geom_hline(yintercept = 0.9, linetype = "dashed", color = "#C44536") +
  geom_vline(xintercept = optimal_dose, linetype = "dashed", color = "#D4A843") +
  annotate("text", x = optimal_dose * 1.1, y = 0.85,
           label = paste0("Dose = ", optimal_dose, " mg\nfor 90% PTA"),
           color = "#D4A843", hjust = 0) +
  labs(title = "Case Study 6: Dose Optimization",
       subtitle = paste0("Target: Cmin > ", target_Cmin, " mg/L, q", tau, "h dosing"),
       x = "Dose (mg)", y = "Probability of Target Attainment") +
  theme_minimal()

```{r}
  theme_minimal()
```

**Interpretation**: Steady state is reached after ~5 half-lives (~14 h). All concentrations remain within the therapeutic range, suggesting this is an appropriate regimen.

## Case Study 3: Antimicrobial PK/PD

### Background

An antibiotic requires $C_{min}/MIC > 1$ and $AUC/MIC > 125$ for efficacy. MIC = 2 mg/L.

```{r}
# PTA analysis for antimicrobial
set.seed(789)
n_sim <- 2000

# Population variability
pop_CL <- 5.0; omega_CL <- 0.25
pop_V  <- 45;  omega_V  <- 0.15
MIC <- 2  # mg/L

# Regimens to compare
regimens <- data.frame(
  Regimen = c("250 mg q12h", "500 mg q12h", "500 mg q8h", "1000 mg q12h"),
  Dose = c(250, 500, 500, 1000),
  Tau  = c(12, 12, 8, 12)
)

# Simulate PTA for each regimen
pta_results <- lapply(1:nrow(regimens), function(r) {
  CL_sim <- pop_CL * exp(rnorm(n_sim, 0, omega_CL))
  V_sim  <- pop_V  * exp(rnorm(n_sim, 0, omega_V))
  ke_sim <- CL_sim / V_sim

  dose <- regimens$Dose[r]
  tau  <- regimens$Tau[r]

  Cmin_ss <- (dose / V_sim) * exp(-ke_sim * tau) / (1 - exp(-ke_sim * tau))
  AUC_ss  <- dose / CL_sim

  data.frame(
    Regimen = regimens$Regimen[r],
    PTA_Cmin = mean(Cmin_ss > MIC),
    PTA_AUC_MIC = mean(AUC_ss / MIC > 125),
    Cmin_median = median(Cmin_ss),
    AUC_MIC_median = median(AUC_ss / MIC)
  )
}) %>% bind_rows()

knitr::kable(pta_results, caption = "PTA Analysis: Antimicrobial PK/PD Targets",
             digits = c(0, 3, 3, 2, 1))
```

## Case Study 4: Delayed PD Response

### Background

A drug shows delayed onset of effect — maximum response occurs after the plasma concentration peak. We model this with an effect compartment.

```{r}
# Simulate PK with effect compartment
mod_ec <- rxode2({
  d/dt(depot)   <- -ka * depot
  d/dt(central) <-  ka * depot - ke * central
  d/dt(effect)  <- ke0 * (central/V - effect)
  Cp <- central / V
  Ce <- effect
  Response <- E0 + (Emax * Ce) / (EC50 + Ce)
})

params_cs4 <- c(ka = 1.2, ke = 0.3, V = 12,
                ke0 = 0.4, E0 = 10, Emax = 80, EC50 = 2)

event_cs4 <- et(time = 0, amt = 200)
times_cs4 <- seq(0, 24, by = 0.1)

sim_cs4 <- rxSolve(mod_ec, params_cs4, event_cs4 %>% et(times_cs4))
cs4_df <- as.data.frame(sim_cs4)

# Plot PK and PD
ggplot(cs4_df, aes(time)) +
  geom_line(aes(y = Cp, color = "Plasma Conc"), linewidth = 1) +
  geom_line(aes(y = Ce, color = "Effect Site Conc"), linewidth = 1, linetype = "dashed") +
  geom_line(aes(y = Response / 5, color = "Response (÷5)"), linewidth = 1.2) +
  scale_y_continuous(
    name = "Concentration (mg/L)",
    sec.axis = sec_axis(~ . * 5, name = "Response")
  ) +
  labs(title = "Case Study 4: Delayed PD Response",
       subtitle = paste0("Effect compartment model, ke0 = ", params_cs4["ke0"]),
       x = "Time (hours)", color = "") +
  scale_color_manual(values = c("#0B6E6B", "#D4A843", "#C44536")) +
  theme_minimal()
```

**Interpretation**: The effect compartment model captures the time delay between plasma concentration and response. The peak effect occurs ~1–2 hours after the plasma peak.

## Case Study 5: Population PK with Covariates

### Background

A population PK study in 80 patients with varying body weight (40–120 kg) and renal function (eGFR: 20–120 mL/min).

```{r}
# Simulate population PK with covariates
set.seed(456)
n_pop <- 80
sparse_times <- c(1, 3, 8)  # 3 samples per subject

pop_data <- expand.grid(ID = 1:n_pop, time = sparse_times) %>%
  mutate(
    WT   = rep(rnorm(n_pop, 70, 15), each = length(sparse_times)),
    eGFR = rep(rnorm(n_pop, 80, 25), each = length(sparse_times)),

    # Covariate effects
    eta_CL = rep(rnorm(n_pop, 0, 0.2), each = length(sparse_times)),
    CL = 5.0 * (WT/70)^0.75 * (eGFR/90)^0.5 * exp(eta_CL),

    eta_V = rep(rnorm(n_pop, 0, 0.15), each = length(sparse_times)),
    V  = 50 * (WT/70)^1.0 * exp(eta_V),

    ke = CL / V,
    conc = (500/V) * exp(-ke * time),
    conc_obs = conc * exp(rnorm(n(), 0, 0.2))
  ) %>%
  arrange(ID, time)

# Visualize covariate effects
p1 <- ggplot(pop_data %>% distinct(ID, .keep_all = TRUE), aes(WT, CL)) +
  geom_point(alpha = 0.6, color = "#0B6E6B") +
  geom_smooth(method = "lm", se = TRUE, color = "#C44536") +
  labs(title = "Weight Effect on Clearance",
       x = "Body Weight (kg)", y = "CL (L/h)") +
  theme_minimal()

p2 <- ggplot(pop_data %>% distinct(ID, .keep_all = TRUE), aes(eGFR, CL)) +
  geom_point(alpha = 0.6, color = "#0B6E6B") +
  geom_smooth(method = "lm", se = TRUE, color = "#C44536") +
  labs(title = "Renal Function Effect on Clearance",
       x = "eGFR (mL/min)", y = "CL (L/h)") +
  theme_minimal()

library(patchwork)
p1 + p2
```

**Interpretation**: Both weight and renal function are significant covariates on CL. Failure to account for them would increase unexplained variability and could lead to suboptimal dosing.

## Case Study 6: Dose Optimization by PTA

### Background

Using the population model from Case Study 5, find the dose that achieves >90% PTA for target $C_{min} > 2$ mg/L.

```{r}
# Dose optimization via simulation
dose_range <- seq(200, 2000, by = 50)
target_Cmin <- 2  # mg/L
tau <- 12  # hours

set.seed(999)
n_sim_opt <- 2000

# Simulate population
WT_sim   <- rnorm(n_sim_opt, 70, 15)
eGFR_sim <- rnorm(n_sim_opt, 80, 25)
eta_CL_sim <- rnorm(n_sim_opt, 0, 0.2)
eta_V_sim  <- rnorm(n_sim_opt, 0, 0.15)

CL_sim <- 5.0 * (WT_sim/70)^0.75 * (eGFR_sim/90)^0.5 * exp(eta_CL_sim)
V_sim  <- 50 * (WT_sim/70)^1.0 * exp(eta_V_sim)
ke_sim <- CL_sim / V_sim

# PTA vs dose
pta_doses <- sapply(dose_range, function(d) {
  Cmin_ss <- (d / V_sim) * exp(-ke_sim * tau) / (1 - exp(-ke_sim * tau))
  mean(Cmin_ss > target_Cmin)
})

df_pta_cs6 <- data.frame(Dose = dose_range, PTA = pta_doses)
optimal_dose <- dose_range[which.min(abs(pta_doses - 0.9))]

ggplot(df_pta_cs6, aes(Dose, PTA)) +
  geom_line(color = "#0B6E6B", linewidth = 1.2) +
  geom_hline(yintercept = 0.9, linetype = "dashed", color = "#C44536") +
  geom_vline(xintercept = optimal_dose, linetype = "dashed", color = "#D4A843") +
  annotate("text", x = optimal_dose * 1.1, y = 0.85,
           label = paste0("Dose = ", optimal_dose, " mg\nfor 90% PTA"),
           color = "#D4A843", hjust = 0) +
  labs(title = "Case Study 6: Dose Optimization",
       subtitle = paste0("Target: Cmin > ", target_Cmin, " mg/L, q", tau, "h dosing"),
       x = "Dose (mg)", y = "Probability of Target Attainment") +
  theme_minimal()
```
