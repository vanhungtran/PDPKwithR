Dataset 1: Single-Dose Oral PK

Code 
library(dplyr)

# Generate the dataset
set.seed(42)
practice1 <- data.frame(
  TIME = c(0, 0.25, 0.5, 1, 2, 3, 4, 6, 8, 12, 24),
  CONC = c(0, 1.2, 3.8, 7.5, 9.2, 8.1, 6.3, 4.1, 2.5, 1.1, 0.3)
)

Exercise A1: NCA

Compute C_{max}, T_{max}, AUC_{0 \to last}, AUC_{0 \to \infty}, and t_{1/2}.

Solution
Code 
nca <- practice1 %>%
  mutate(log_conc = log(CONC))

# Cmax
cmax <- max(nca$CONC)
tmax <- nca$TIME[which.max(nca$CONC)]

# AUC (linear trapezoidal)
auc_last <- sum(diff(nca$TIME) * (nca$CONC[-1] + nca$CONC[-nrow(nca)]) / 2)

# Terminal slope (last 4 points)
term_idx <- 8:11
lm_term <- lm(log_conc ~ TIME, data = nca[term_idx, ])
ke <- -coef(lm_term)[2]
t_half <- log(2) / ke

# AUC∞
auc_inf <- auc_last + nca$CONC[nrow(nca)] / ke

c(Cmax = cmax, Tmax = tmax, AUC_last = auc_last,
  AUC_inf = auc_inf, t_half = t_half, ke = ke)
        Cmax         Tmax     AUC_last AUC_inf.TIME  t_half.TIME      ke.TIME 
    9.200000     2.000000    60.400000    62.551679     4.971435     0.139426

Dataset 4: Population PK

Code 
set.seed(555)
n_subj <- 8
id_times <- c(0.5, 1, 2, 4, 8, 12, 24)

practice4 <- expand.grid(ID = 1:n_subj, TIME = id_times) %>%
  mutate(
    DOSE = 500,
    CL = 5 * exp(rep(rnorm(n_subj, 0, 0.25), each = length(id_times))),
    V  = 50 * exp(rep(rnorm(n_subj, 0, 0.15), each = length(id_times))),
    ke = CL / V,
    CONC = (DOSE / V) * exp(-ke * TIME),
    CONC = CONC * exp(rnorm(n(), 0, 0.1))
  ) %>%
  arrange(ID, TIME)

Exercise A4: Population Analysis

Compute individual k_e via log-linear regression for each subject. Summarize mean and SD of k_e and V.

Solution
Code 
indiv_params <- practice4 %>%
  group_by(ID) %>%
  summarise(
    ke = -coef(lm(log(CONC) ~ TIME))[2],
    C0 = exp(coef(lm(log(CONC) ~ TIME))[1]),
    V = 500 / C0,
    .groups = "drop"
  )

indiv_params %>%
  summarise(
    ke_mean = mean(ke), ke_sd = sd(ke),
    V_mean = mean(V), V_sd = sd(V)
  )

Exercise: Dose Optimization

Using a population model with CL = 5 L/h (CV = 25%), V = 50 L (CV = 15%), determine the dose needed for 90% PTA with target C_{min} > 1 mg/L at q12h dosing.

Solution
Code 
set.seed(999)
n_sim <- 5000
CL_sim <- 5 * exp(rnorm(n_sim, 0, 0.25))
V_sim  <- 50 * exp(rnorm(n_sim, 0, 0.15))
ke_sim <- CL_sim / V_sim
tau <- 12
target <- 1

# Try different doses
for (d in seq(200, 2000, by = 10)) {
  Cmin <- (d / V_sim) * exp(-ke_sim * tau) / (1 - exp(-ke_sim * tau))
  pta <- mean(Cmin > target)
  if (pta >= 0.9) {
    cat("Dose for 90% PTA:", d, "mg\n")
    break
  }
}
Dose for 90% PTA: 220 mg
